\section{Plotting}
\label{sec:plotting}


\subsection{Multi}

\begin{verbatim}
Each series handled as a separate element, e.g.,ER(N=1000,ad=10.0,b=true)ER(N=1000,ad=15.0,b=true)ER(N=1000,ad=20.0,b=true)
\end{verbatim}



\subsection{Single}

\begin{verbatim}
Each series[] is combined as an elementDifferent parameter is used as value on the x-axis, e.g.,ER(N=1000,ad=X,b=true) => x := Average DegreeER(N=X,ad=25.0,b=true) => x := Nodes
\end{verbatim}




\subsection{Single-by}

\begin{verbatim}
Each series[] is combined as an elementSpecified parameter is used as value on the x-axis, e.g.,DegreeDistribution - “EDGES”ClusteringCoefficient - “CLUSTERING_COEFFICIENT”
\end{verbatim}

\end{verbatim}
